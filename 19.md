> 内容包括vue.js + vue-router + vuex + axios

### 认识axios

  npm install -- save aixos

>  axios是一个http库， 主要用于向后端发送和请求数据，它主要配合vue.js使用。相当于原生js里面的XMLHttpRequest对象。简写做xhr，我们也叫做ajax库，也相当于jQuery里面的ajax函数

####  了解es6的Promise

~~~
  new Promise(function (reslove, reject){
    let ok = 'test'
    if(ok === 'test') {
      reslove('执行成功')
    } else {
      reject('失败')
    }
    }).then( params => {
      // 执行成功
      // params === '执行成功'
      console.log(params)
    }).catch( err => {
      // 执行失败
      // err === '执行失败'
      console.log(err)
    })
~~~
